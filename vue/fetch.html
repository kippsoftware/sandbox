Fetch

function log(data) {
  document.getElementById("log").innerHTML = data;
}

fetch('table.json')
  .then(response => response.json())
  .then(data => log(JSON.stringify(data, undefined, 2)));

fetch(url) returns a Promise

.then(f) calls f with the response; e.g., .json(), .text(), .blob(), which returns a Promise to complete parsing

.then(f) calls f with the data; if json(), then a JSON object
